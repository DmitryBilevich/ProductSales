CREATE PROCEDURE [dbo].[ClearProductImportStaging]
    @ImportSessionID UNIQUEIDENTIFIER
AS
BEGIN
    SET NOCOUNT ON;
    
    DELETE FROM ProductImportStaging 
    WHERE ImportSessionID = @ImportSessionID;
    
    SELECT 
        @@ROWCOUNT as ClearedCount,
        'Staging data cleared successfully' as Message;
END
GO
